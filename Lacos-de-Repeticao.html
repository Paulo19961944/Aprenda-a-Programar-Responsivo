// Laço de Repetição for

#include < stdio.h >

#include < locale.h >

int cont;

int main()

{

// Contagem Crescente

for(cont=0;cont<=10;cont=cont+1){

printf("%d\n", cont);

}

// Contagem Decrescente

for(cont=9;cont>=0;cont=cont-1){

printf("%d\n", cont);

}

return 0;

}
